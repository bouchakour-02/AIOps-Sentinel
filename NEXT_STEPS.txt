================================================================================
  INTEGRATION EXAMPLES
================================================================================

With Elasticsearch:
   from elasticsearch import Elasticsearch
   es = Elasticsearch(["http://localhost:9200"])
   response = es.search(index="vermeg-logs", size=50)
   
   # Get logs and predict anomalies
   for hit in response['hits']['hits']:
       log = hit['_source']['Body']
       pred = requests.post(
           "http://localhost:8000/api/predict/log-anomaly",
           json={"message": log}
       )
       if pred.json()["anomaly"]:
           print(f"🚨 Anomaly: {log}")

With Prometheus:
   # Get metrics and check for anomalies
   cpu = get_metric("node_cpu_percent")
   memory = get_metric("node_memory_percent")
   
   pred = requests.post(
       "http://localhost:8000/api/predict/metric-anomaly",
       json={"cpu": cpu, "memory": memory, "db_errors": 0}
   )
   if pred.json()["anomaly"]:
       print(f"⚠️ Metric anomaly detected")
